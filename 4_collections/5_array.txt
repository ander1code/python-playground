	Accessing a range of array items in Python (slicing operation [:])
	
		Exemplo 01:
		
			import array as arr

			arr_int = arr.array("i",[2,3,7,4,1,5,6,7,5,9])

			print(arr_int[:]) # todos
			print(arr_int[:-1]) # sem o ultimo
			print(arr_int[0:3]) # 3 primeiros
			print(arr_int[-1:]) # ultimo item
			print(arr_int[-3:]) # 3 ultimos
			print(arr_int[:-3]) # todos sem os 3 ultimos

			import array

Python - Sort Arrays

	Sort Arrays Using a Sorting Algorithm
	
		Example 01:
		
			import array as arr
			a = arr.array('i', [10,5,15,4,6,20,9])

			def sort_array(order="desc", array=[]):
				for i in range(0,len(a)):
					for j in range(i+1, len(a)):
						if order=="desc":
							if a[i] < a[j]:
								temp = a[i]
								a[i] = a[j]
								a[j] = temp
						else:
							if a[i] < a[j]:
								temp = a[i]
								a[i] = a[j]
								a[j] = temp

			sort_array(order="desc", array=a)
			print(a)
			sort_array(order="asc", array=a)
			print(a)
